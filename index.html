  <script>

    const truths = [

      "What's the craziest thing you've done for love?",

      "Have you ever lied to your partner? If yes, about what?",

      "What’s something your partner does that melts your heart?",

      "What's a secret you've never told anyone?",

      "What was your first impression of your partner?",

      "Would you rather lose your phone or lose your partner for a day?",

      "What's your most embarrassing love story?",

      "When did you realize you were in love with them?"

    ];


    const dares = [

      "Kiss your partner for 5 seconds.",

      "Hug your partner for 1 full minute.",

      "Whisper something naughty into their ear.",

      "Do a romantic dance together with no music.",

      "Act like you're proposing with full drama.",

      "Send your partner a random flirty message.",

      "Compliment your partner using only movie quotes.",

      "Draw a heart on your partner's hand with your finger."

    ];


    const whoMostly = [

      "Who mostly says 'I love you' first?",

      "Who mostly gets jealous?",

      "Who mostly gives in after an argument?",

      "Who mostly plans the date nights?",

      "Who mostly takes selfies?",

      "Who mostly gets emotional while watching movies?",

      "Who mostly initiates deep conversations?",

      "Who mostly sends good morning texts first?"

    ];


    function showRandom(type) {

      let output = document.getElementById("output");

      let list;

      if (type === 'truth') list = truths;

      else if (type === 'dare') list = dares;

      else if (type === 'who') list = whoMostly;


      const random = list[Math.floor(Math.random() * list.length)];

      output.innerText = random;

    }


    function showDrink() {

      document.getElementById("output").innerText = "Take a sip and smile 🍷💕";

    }

  </script>
